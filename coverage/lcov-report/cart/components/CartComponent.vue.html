  methods: {
    toggleCart() {
      this.$store.commit("cart/TOGGLE_CART");

    closeCart() {
      this.$store.commit("cart/CLOSE_CART");
    },
    clearCart() {
      this.$store.commit("cart/CLEAR_CART");
    },
    handleEscape(event) {
      if (event.key === "Escape" && this.isCartOpen === true) {
        this.closeCart();
      }
